698:+// Returns [p1, exitStub1, ...midCorners, exitStub2, p2].
699:+// If waypoints[0]._corner is true, uses those as explicit corners directly.
700:+// Otherwise applies smart-exit routing: flips the stub if it points AWAY from
701:+// the target, so the wire goes toward the destination instead of U-turning.
702:+function computeWireOrthoPoints(p1, e1, e2, p2, waypoints = []) {
703:+  // Explicit corner mode — stored by segment dragging
704:+  if (waypoints.length > 0 && waypoints[0]._corner) {
705:+    const pts = [p1, ...waypoints, p2];
706:+    return pts.filter((pt, i, arr) => i === 0 || pt.x !== arr[i - 1].x || pt.y !== arr[i - 1].y);
707:+  }
708:+
709:+  // Smart exit: flip stub when it points away from the other endpoint
710:+  const dx1 = e1.x - p1.x, dy1 = e1.y - p1.y;
711:+  const dx2 = e2.x - p2.x, dy2 = e2.y - p2.y;
712:+  const e1IsVert = Math.abs(dy1) > Math.abs(dx1);
713:+  const e2IsVert = Math.abs(dy2) > Math.abs(dx2);
714:+
715:+  let se1 = e1, se2 = e2;
716:+  if (e1IsVert) {
717:+    if (dy1 !== 0 && (p2.y - p1.y) * dy1 < 0) se1 = { x: p1.x, y: p1.y - dy1 };
718:+  } else {
719:+    if (dx1 !== 0 && (p2.x - p1.x) * dx1 < 0) se1 = { x: p1.x - dx1, y: p1.y };
720:+  }
721:+  if (e2IsVert) {
722:+    if (dy2 !== 0 && (p1.y - p2.y) * dy2 < 0) se2 = { x: p2.x, y: p2.y - dy2 };
723:+  } else {
724:+    if (dx2 !== 0 && (p1.x - p2.x) * dx2 < 0) se2 = { x: p2.x - dx2, y: p2.y };
725:+  }
726:+
727:+  const sdx1 = se1.x - p1.x, sdy1 = se1.y - p1.y;
728:+  const sdx2 = se2.x - p2.x, sdy2 = se2.y - p2.y;
729:+  const e1Horiz = Math.abs(sdx1) >= Math.abs(sdy1);
730:+  const e2Horiz = Math.abs(sdx2) >= Math.abs(sdy2);
731:+
732:+  let midPts;
733:+  if (e1Horiz && e2Horiz) {
734:+    const midX = (se1.x + se2.x) / 2;
735:+    midPts = [{ x: midX, y: se1.y }, { x: midX, y: se2.y }];
736:+  } else if (!e1Horiz && !e2Horiz) {
737:+    const midY = (se1.y + se2.y) / 2;
738:+    midPts = [{ x: se1.x, y: midY }, { x: se2.x, y: midY }];
739:+  } else if (e1Horiz && !e2Horiz) {
740:+    midPts = [{ x: se2.x, y: se1.y }];
741:+  } else {
742:+    midPts = [{ x: se1.x, y: se2.y }];
743:+  }
744:+
745:+  let pts = [p1, se1, ...midPts, se2, p2];
746:+  return pts.filter((pt, i, arr) => i === 0 || pt.x !== arr[i - 1].x || pt.y !== arr[i - 1].y);

749:+// ─── SINGLE SOURCE OF TRUTH: full orthogonal point list for any wire mode ──
750:+// Mode 1 – explicit corners (_corner:true, from segment dragging): use points as-is.
751:+// Mode 2 – route-hint waypoints (clicked mid-draw, no _corner): midX dog-leg.
752:+// Mode 3 – no waypoints: smart-exit auto-routing.
753:+function getWirePoints(p1, e1, e2, p2, waypoints = []) {
754:+  // Mode 1: explicit corners stored by segment dragging
755:+  if (waypoints.length > 0 && waypoints[0]._corner) {
756:+    let pts = [p1, ...waypoints, p2];
757:+    return pts.filter((pt, i, arr) => i === 0 || pt.x !== arr[i - 1].x || pt.y !== arr[i - 1].y);
758:+  }
759:+
760:+  // Mode 2: route-hint waypoints – midX dog-leg between each successive hint
761:+  if (waypoints.length > 0) {
762:+    const hints = [e1, ...waypoints, e2];
763:+    let pts = [p1];
764:+    for (let i = 0; i < hints.length - 1; i++) {
765:+      const a = hints[i], b = hints[i + 1];

769:+      pts.push({ x: midX, y: b.y });

772:+    return pts.filter((pt, i, arr) => i === 0 || pt.x !== arr[i - 1].x || pt.y !== arr[i - 1].y);
773:+  }
774:+
775:+  // Mode 3: no waypoints – smart-exit routing
776:+  return computeWireOrthoPoints(p1, e1, e2, p2, []);
